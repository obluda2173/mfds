function forward_substitution(L, b)
    n = size(L, 1)
    y = zeros(Float64, n)

    for i in 1:n
        sum = b[i]

        for j in 1:(i-1)
            sum -= L[i, j] * y[j]
        end

        y[i] = sum / L[i, i]
    end
    return y
end

function backward_substitution(U, y)
    n = size(U, 1)
    x = zeros(Float64, n)

    for i in n:-1:1
        sum = y[i]

        for j in (i+1):n
            sum -= U[i, j] * x[j]
        end
        x[i] = sum / U[i, i]
    end
    return x
end

L = [ 1  0  0;

     -1  2  1]

U = [ 2  1 -1;
      0  3 -1;
      0  0 -1]

      7;
      3]

y = forward_substitution(L, b)
x = backward_substitution(U, y)

** Block B: Theory & Proof Memorization
*** Task
Learn and memorize the Steinitz Exchange Theorem (Theorem III.1.47).

*** Solution

Let $V$ be a vector space over a field $\mathbb{F}$. Let $L = \left\{ v_1, v_2, \ldots, v_m \right\}$ be linearly independent and $S = \left\{ w_1, w_2, \ldots, w_{n} \right\}$ be a spanning set over $V$. Then $m \le n$, and there exists $S' \subseteq S$ with $|S'| = n - m$ such that $S' \cup L$ spans $V$.

*Proof.*

*Base Case (m = 0).* $L = \emptyset$. Then, $m = 0 \leq n$, and $L \cup S = S$ spans $V$.

*Inductive Step.* Assume the statement holds for $m - 1$. That is, applying the theorem to $\left\{ v_1, v_2, \ldots, v_{m-1} \right\}$ and $\left\{ w_1, \ldots, w_{n} \right\}$ there exists $S_{m-1} \subseteq S$ with $|S_{m-1}| = n - (m-1)$ such that

$$\left\{ v_1, \ldots, v_{m-1} \right\} \cup S_{m-1}$$

spans $V$. In particular $m-1 \leq n$, so this set is well defined and non empty.

a. Define the terms kernel, image, and rank of a linear map. (2 Points)

b. Provide a proper formulation for the Steinitz exchange theorem (Theorem III.1.47). (2 Points)

c. Give a proof of the Steinitz exchange theorem. (6 Points)
#+end_quote

*** Solution
**** (a)

Let $T : V \rightarrow W$ be a linear map between vector spaces $V$ and $W$ over a field $\mathbb{F}$.

- *Kernel of $T$:*

  $$ker(T) = \left\{ v \in V : T(v) = 0_{W} \right\}$$

  A kernel is the preimage of a zero vector in $W$. It is a subspace of $V$.

- *Image of $T$:*

  $im(T) = \left\{ T(v) : v \in V \right\} = T(V)$

  The image is the set of all outputs of $T$. It is a subspace of $W$.

- *Rank of $T$:*

  $$rank(T) = dim(im(T))$$

  Rank is the dimension of the image space.
